puppet.conf

默认$vardir为/var/lib/puppet

[main]
	logdir = /var/log/puppet #默认日志存放路径
	rundir = /var/run/puppet #pid存放路径
	ssldir = $vardir/ssl #证书存放目录
[agent]
	classfile = $vardir/classes.txt
	localconfig = $vardir/localconfig
	server = puppet.domain.com #设置agent认证连接master端的服务器名称
[master]
